[docs]    def run(self):
        s = socket.socket()

        port = 8000
        s.bind(('', port))

        while True:
            print('listening')
            s.listen(5)
            c, addr = s.accept()
            print('receiving')
            
            queue = Queue.Queue()
            while True:
                data = c.recv(4096)
                print(data)
                for d in data.split("\n"):
                    if d:
                        print(d)
                        queue.put(list(map(int, d.split(","))))
                        
                try:
                    if not queue.empty():
                        x, y = queue.get()
                        self.to_servo(x, y)
                except:
                    break
            c.close()
        s.close()
